edge=get.edgelist(rfid)
id1=edge[,1];id2=edge[,2] #id1id2
time=E(rfid)$Time
time=as.POSIXlt("2010-12-06 13:00:00")+time
hour=strftime(time,"%H")

period=numeric(length(hour))  
period[hour %in% 6:13]=1    #1:
period[hour %in% 14:21]=2   #2:
period[hour %in% c(0:5,22:23)]=3 #3:

n=vcount(rfid)
type=factor(V(rfid)$Status)
R=matrix(0,n,length(levels(type)))
R[cbind(1:n,type)]=1
CONTACT=array(0,c(n,n,3)) #CONTACT:
for(i in 1:ecount(rfid))  CONTACT[id1[i],id2[i],period[i]]=CONTACT[id1[i],id2[i],period[i]]+1
CONTACT=CONTACT/3 #
FREQ=CONTACT>0 #FREQ:

CONTACT.Status=function(TYPE=c("",""),PERIOD=c("","","")){
  if (TYPE=="") M=FREQ else if (TYPE=="") M=CONTACT
  i = PERIOD==c("","","")
  ANS=t(R)%*%M[,,i]%*%R
  ANS=ANS+t(ANS)-diag(diag(ANS))
  rownames(ANS)=colnames(ANS)=levels(type)
  return(ANS)
}
