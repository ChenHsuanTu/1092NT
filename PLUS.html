計算該醫院內各類型人員在各時段(白天、晚上、深夜)的相互接觸型態(接觸人數與時間)。並討論當發生院內感染時 (e.g. COVID-19)，如何依照接觸型態找出「超級接觸者」，進行隔離措施。 

library(igraph)
options(digits=0)
load("D:/1092NT/Data/Task1.RData")

edge=get.edgelist(rfid)
id1=edge[,1];id2=edge[,2] #資料中id1都小於id2
time=E(rfid)$Time
time=as.POSIXlt("2010-12-06 13:00:00")+time
hour=strftime(time,"%H")

period=numeric(length(hour))  
period[hour %in% 6:13]=1    #1:白天
period[hour %in% 14:21]=2   #2:晚上
period[hour %in% c(0:5,22:23)]=3 #3:深夜

n=vcount(rfid)
type=factor(V(rfid)$Status)
R=matrix(0,n,length(levels(type)))
R[cbind(1:n,type)]=1
CONTACT=array(0,c(n,n,3)) #CONTACT:人與人接觸時間
for(i in 1:ecount(rfid))  CONTACT[id1[i],id2[i],period[i]]=CONTACT[id1[i],id2[i],period[i]]+1
CONTACT=CONTACT/3 #以分鐘為單位
FREQ=CONTACT>0 #FREQ:接觸人次

CONTACT.Status=function(TYPE=c("接觸人數","接觸時間"),PERIOD=c("白天","晚上","深夜")){
  if (TYPE=="接觸人數") M=FREQ else if (TYPE=="接觸時間") M=CONTACT
  i = PERIOD==c("白天","晚上","深夜")
  ANS=t(R)%*%M[,,i]%*%R
  ANS=ANS+t(ANS)-diag(diag(ANS))
  rownames(ANS)=colnames(ANS)=levels(type)
  return(ANS)
}
