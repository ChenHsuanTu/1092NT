Centrality & Subgroup

CS = function(Gx){
  for(i in 1:components(Gx)$no){
    G = delete.vertices(Gx,which(components(Gx)$member!=i))
    D = degree(G)
    C = closeness(G)
    B = betweenness(G)
    EC = eigen_centrality(G)$vector
    auth = authority_score(G)$vector
    hub = hub_score(G)$vector
    PR = page_rank(G)$vector
    DF = data.frame(Node=names(V(G)),D=D,C=C,B=B,auth=auth,hub=hub,PR=PR)
    if(i==1) Data=DF
    else Data=rbind(Data,DF)
  }
  Data[order(as.numeric(Data$Node)),]
  Data$component=components(Gx)$membership
  Gu=as.undirected(Gx)
  Data$k_core=coreness(Gu)
  Data$GN=cluster_edge_betweenness(Gu)$membership
  Data
}
